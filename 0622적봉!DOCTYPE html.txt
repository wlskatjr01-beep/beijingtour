    <script>
        function openDay(evt, dayName) {
            var i, daycontent, tablinks;
            
            // 모든 일정 콘텐츠 숨김
            daycontent = document.getElementsByClassName("day-content");
            for (i = 0; i < daycontent.length; i++) {
                daycontent[i].classList.remove("active");
            }

            // 모든 탭 버튼 비활성화
            tablinks = document.getElementsByClassName("tab-btn");
            for (i = 0; i < tablinks.length; i++) {
                tablinks[i].classList.remove("active");
            }

            // 선택된 일정 보여주기 및 클릭된 버튼 활성화
            document.getElementById(dayName).classList.add("active");
            evt.currentTarget.classList.add("active");

            // 선택 시 모바일 화면이 상단으로 자동 스크롤되도록 지원
            window.scrollTo({
                top: document.querySelector('.tab-container').offsetTop - 10,
                behavior: 'smooth'
            });
        }
    </script>
